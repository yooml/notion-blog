发现是subPages := page.GetSubPages()获取到的子页面不包含引用页面的原因
查看此方法的实现：
// GetSubPages return list of ids for direct sub-pages of this page
func (p *Page) GetSubPages() []string {
	root := p.Root()
	panicIf(!isPageBlock(root))
	subPages := map[string]struct{}{}
	seenBlocks := map[string]struct{}{}
	blocksToVisit := append([]string{}, root.ContentIDs...)
	for len(blocksToVisit) > 0 {
		id := ToDashID(blocksToVisit[0])
		blocksToVisit = blocksToVisit[1:]
		if _, ok := seenBlocks[id]; ok {
			continue
		}
		seenBlocks[id] = struct{}{}
		block := p.BlockByID(id)
		if p.IsSubPage(block) {
			subPages[id] = struct{}{}
		}
		// need to recursively scan blocks with children
		blocksToVisit = append(blocksToVisit, block.ContentIDs...)
	}
	res := []string{}
	for id := range subPages {
		res = append(res, id)
	}
	sort.Strings(res)
	return res
}

原因找到：
if p.IsSubPage(block) {
			subPages[id] = struct{}{}
		}
page.GetSubPages()内部会判断此block是不是子页，而超链引用的话就不是子页就不会添加进队列，所以只能自己改了
重写GetSubPages改为不判断是否是block后，出现了所有子block都被下载下来的情况，并且产生大量原子block的html文件，如图:
修改为：
		block := p.BlockByID(id)
		page, _ := d.DownloadPage(notionapi.ToNoDashID(id))
		bblocksToVisit := append([]string{}, page.Root().ContentIDs...)
		if len(bblocksToVisit)!=0{
			log.Println(id)
			subPages[id] = struct{}{}
		}
DownloadPage还是会下载每个block，但是生成多余html文件的问题解决
思考：此方法每个block都要去下载，因为需要获取page对象，并且通过page.Root().ContentIDs的数量判断这个page是引用还是只是block，如果是引用那么len(bblocksToVisit)不为零，从而去执行将page转化为html静态文件的操作，如果是原子的block，那么就不会将其转化为html。
解决方案：
if ifQuota(d,id){
			subPages[id] = struct{}{}
		}

func ifQuota(d *caching_downloader.Downloader,p string) bool {
	records,err:=d.Client.GetBlockRecords([]string{p})
	if err!=nil{
		log.Println(err)
	}

	for _, results := range records.RawJSON{
		for _,result := range results.([]interface{}){
			map_result:=result.(map[string]interface{})
			if value, ok := map_result["value"]; ok {
				map_value := value.(map[string]interface{})
				if content, ok := map_value["content"]; ok {
					if len(content.([]interface{}))!=0{
						log.Println("子模块(content)个数:",len(content.([]interface{})))
						return true
					}
				}
			}
		}
	}
	return false
}
Client.GetBlockRecords其实是请求官方接口“/api/v3/getRecordValues“的封装，解决方法就是改为每个block请求一下判断是否是引用，再进行下载page的操作
